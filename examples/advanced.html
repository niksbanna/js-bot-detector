Custom Signal Example

    
import { BotDetector, Signal } from '@niksbanna/bot-detector';

// Create a custom signal
class CustomBotSignal extends Signal {
  static id = 'custom-bot-check';
  static category = 'custom';
  static weight = 0.8;
  static description = 'Custom bot detection logic';
  
  async detect() {
    // Your custom detection logic
    const suspicious = window.myBotFlag === true;
    return this.createResult(suspicious, { checked: true }, 0.9);
  }
}

// Use with detector
const detector = new BotDetector();
detector.registerSignal(new CustomBotSignal());
const result = await detector.detect();
